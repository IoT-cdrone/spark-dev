  parseErrors: (raw) ->
    lines = raw.split "\n"
    errors = []
    for line in lines
      result = line.match /^([^:]+):(\d+):(\d+):\s(\w+):(.*)$/
      if result and result[4] == 'error'
        errors.push {
          file: result[1],
          row: result[2],
          col: result[3],
          type: result[4],
          message: result[5]
        }
    errors
